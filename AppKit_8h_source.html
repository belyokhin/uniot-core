93 virtual void registerWithBus(CoreEventBus &eventBus) override {

95 eventBus.openDataChannel(unLisp::Channel::OUT_LISP, 5);
96 eventBus.openDataChannel(unLisp::Channel::OUT_LISP_LOG, 10);

98 eventBus.openDataChannel(unLisp::Channel::OUT_EVENT, 10);

100 eventBus.registerEntity(&Date::getInstance());
101 eventBus.registerEntity(&mNetwork);
102 eventBus.registerEntity(&mMQTT);
103 eventBus.registerEntity(&getLisp());
104 eventBus.registerEntity(&mLispDevice);

106 ->listenToEvent(NetworkScheduler::Topic::CONNECTION)
107 ->listenToEvent(MQTTKit::Topic::CONNECTION));
108
109 if (mpNetworkDevice) {

112 UNIOT_LOG_WARN("Configure Network Controller before registering to the event bus");
113 }
114 }

218 inline void _initListeners() {
219 mpNetworkEventListener = MakeUnique<CoreCallbackEventListener>([&](int topic, int msg) {
220 if (NetworkScheduler::CONNECTION == topic) {
221 switch (msg) {
222 case NetworkScheduler::SUCCESS:
223 UNIOT_LOG_DEBUG("AppKit Subscriber, SUCCESS, ip: %s", WiFi.localIP().toString().c_str());
224 break;
225 case NetworkScheduler::ACCESS_POINT:
226 UNIOT_LOG_DEBUG("AppKit Subscriber, ACCESS_POINT");
227 mpNetworkEventListener->receiveDataFromChannel(NetworkScheduler::Channel::OUT_SSID, [this](unsigned int id, bool empty, Bytes data) {
228 if (!empty) {
229 UNIOT_LOG_DEBUG("SSID: %s", data.terminate().c_str());
230 }
231 });
232 break;
233
234 case NetworkScheduler::CONNECTING:
235 UNIOT_LOG_DEBUG("AppKit Subscriber, CONNECTING");
236 mpNetworkEventListener->receiveDataFromChannel(NetworkScheduler::Channel::OUT_SSID, [this](unsigned int id, bool empty, Bytes data) {
237 if (!empty) {
238 UNIOT_LOG_DEBUG("SSID: %s", data.terminate().c_str());
239 }
240 });
241 break;
242
243 case NetworkScheduler::DISCONNECTED:
244 UNIOT_LOG_DEBUG("AppKit Subscriber, DISCONNECTED");
245 break;
246
247 case NetworkScheduler::FAILED:
248 default:
249 UNIOT_LOG_DEBUG("AppKit Subscriber, FAILED");
250 break;
251 }
252 return;
253 }
254 if (MQTTKit::CONNECTION == topic) {
255 switch (msg) {
256 case MQTTKit::SUCCESS:
257 UNIOT_LOG_DEBUG("AppKit Subscriber, MQTT SUCCESS");
258 mMQTT.renewSubscriptions();
259 break;
260 case MQTTKit::FAILED:
261 default:
262 UNIOT_LOG_DEBUG("AppKit Subscriber, MQTT FAILED");
263 break;
264 }
265 return;
266 }
267 });
268 }
