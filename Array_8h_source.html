57 Array(size_t size, const T* values) : mData(nullptr), mSize(0), mCapacity(0) {
58 if (values && size > 0) {
59 mData = new (std::nothrow) T[size];
60 if (!mData) {
61 return;
62 }

64 if constexpr (std::is_trivially_copyable<T>::value) {

214 bool reserve(size_t newCapacity) {
215 if (newCapacity <= mCapacity) {
216 return true;
217 }
218
219 T* new_data = new (std::nothrow) T[newCapacity];
220 if (!new_data) {
221 return false;
222 }
223
224 // Move existing elements
225 for (size_t i = 0; i < mSize; ++i) {
226 new_data[i] = std::move(mData[i]);
227 }
228
229 // Initialize new elements
230 if constexpr (std::is_trivially_default_constructible<T>::value) {
231 memset(reinterpret_cast<void*>(new_data + mSize), 0, sizeof(T) * (newCapacity - mSize));
232 } else {
233 for (size_t i = mSize; i < newCapacity; ++i) {
234 new_data[i] = T();
235 }
236 }
237
238 delete[] mData;
239 mData = new_data;
240 mCapacity = newCapacity;
241 return true;
242 }
