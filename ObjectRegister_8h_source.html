38 bool link(const String &name, RecordPtr link, uint32_t id = FOURCC(____)) {

41
42 template <typename T>
43 T *get(const String &name, size_t index) {
44 Pair<uint32_t, RecordPtr> record;
45 if (getRegisterValue(name, index, record)) {
46 if (!record.second) {
47 return nullptr;
48 }
49 if (ObjectRegisterRecord::exists(record.second)) {
50 return Type::safeStaticCast<T>(record.second);
51 }
52 setRegisterValue(name, index, MakePair(FOURCC(dead), nullptr));
53 UNIOT_LOG_DEBUG("record is dead [%s][%d]", name.c_str(), index);
54 }
55 return nullptr;
56 }
57};
