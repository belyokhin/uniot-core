37template <typename T>
38class Register {
39 public:
40 using IteratorCallback = std::function<void(const String&, SharedPointer<Array<T>>)>;

59 bool setRegister(const String& name, const T* values, size_t count) {
60 if (!values && count > 0) {
61 return false;
62 }
63
64 mRegisterMap.remove(name);
65
66 if (count > 0) {
67 auto newArray = MakeShared<Array<T>>(count, values);
68 if ((*newArray).size() != count) {
69 return false;
70 }
71 mRegisterMap.put(name, std::move(newArray));
72 for (size_t i = 0; i < count; ++i) {
73 _processRegister(name, values[i]);
74 }
75 }
76
77 return true;
78 }
79
87 bool addToRegister(const String& name, const T& value) {
88 SharedPointer<Array<T>> reg = mRegisterMap.get(name, nullptr);
89 if (!reg) {
90 reg = MakeShared<Array<T>>(4); // Default capacity of 4
91 mRegisterMap.put(name, reg);
92 }
93 if (reg->push(value)) {
94 _processRegister(name, value);
95 return true;
96 }
97 return false;
98 }
99
108 bool getRegisterValue(const String& name, size_t idx, T& outValue) const {
109 auto reg = mRegisterMap.get(name, nullptr);
110 if (reg) {
111 return reg->get(idx, outValue);
112 }
113 return false;
114 }
115
124 bool setRegisterValue(const String& name, size_t idx, const T& value) {
125 auto reg = mRegisterMap.get(name, nullptr);
126 if (reg && reg->set(idx, value)) {
127 _processRegister(name, value);
128 return true;
129 }
130 return false;
131 }
132
139 size_t getRegisterLength(const String& name) const {
140 auto reg = mRegisterMap.get(name, nullptr);
141 if (reg) {
142 return reg->size();
143 }
144 return 0;
145 }
146
152 void iterateRegisters(IteratorCallback callback) const {
153 if (!callback) return;
154
155 mRegisterMap.begin();
156 while (!mRegisterMap.isEnd()) {
157 auto& item = mRegisterMap.current();
158 callback(item.first, item.second);
159 mRegisterMap.next();
160 }
161 }
162
163 protected:
170 virtual void _processRegister(const String& name, const T& value) {}
171
172 private:
173 // Map from register name (String) to Array<T>
174 Map<String, SharedPointer<Array<T>>> mRegisterMap;
175};
